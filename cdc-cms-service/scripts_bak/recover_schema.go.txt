package main

import (
"context"
"fmt"
"gorm.io/driver/postgres"
"gorm.io/gorm"
"bytes"
"net/http"
"encoding/json"
"os"
)

func main() {
	dsn := "host=localhost user=admin password=admin dbname=centralized_data_service port=5432 sslmode=disable"
	db, err := gorm.Open(postgres.Open(dsn), &gorm.Config{})
	if err != nil {
		fmt.Println("DB error:", err)
		return
	}

	var sourceID *string
	err = db.Raw("SELECT airbyte_source_id FROM table_registries WHERE source_table IN ('identitycounters', 'payment-bill-codes') LIMIT 1").Scan(&sourceID).Error
	if err != nil || sourceID == nil {
		fmt.Println("Could not find source ID", err)
		return
	}

	fmt.Println("Found Source ID:", *sourceID)
	
	url := fmt.Sprintf("http://localhost:8001/api/v1/sources/discover_schema")
	payload, _ := json.Marshal(map[string]string{"sourceId": *sourceID, "disable_cache": "true"})
	req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
	req.Header.Set("Content-Type", "application/json")
	
	// Default Airbyte auth if any, but default local airbyte uses auth ? maybe no auth or airbyte/password.
	// Oh Airbyte API might require basic auth. 
	req.SetBasicAuth("airbyte", "password")
	
	client := &http.Client{}
	resp, err := client.Do(req)
	if err != nil {
		fmt.Println("Airbyte API error:", err)
		return
	}
	defer resp.Body.Close()
	fmt.Println("Airbyte status:", resp.Status)
}
